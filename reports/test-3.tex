\begin{equation}
  E=mc^{2}
\end{equation}
で関係付けられる。
\par
$c$は光速で、
\begin{equation}
  \label{speed-of-light}
  c=299{.}792{.}458 \, \mathrm{m/s}
\end{equation}
である。
\par
数式番号をつけないときは
\[
  1+1=2
\]
もしくは
\begin{equation*}
  1+2=3
\end{equation*}
でできる.
\par
下付き文字はこう！
\begin{equation}
  e_{i}
\end{equation}
いぇーい
\par

  \frac{\pi}{2}=
  \left(\int_{0}^{\infty} \frac{\sin x}{\sqrt{x}} dx \right)^2=
  \sum_{k=0}^{\infty} \frac{(2k)!}{2^{2k}(k!)^2} \frac{1}{2k+1}=
  \prod_{k=1}^{\infty} \frac{4k^2}{4k^2-1}+
  \sum_{j=1}^{\infty} \frac{4j^2 + 2}{3}
\end{equation}
